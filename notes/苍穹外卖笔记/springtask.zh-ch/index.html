Spring Task

一、使用步骤

  1. 导入Maven坐标（一般包含在starter里面）
  2. 启动类添加注解@EnableScheduling
  3. 自定义定时任务类（要加@Component注解）

二、使用示例

 1
 2@Component
 3@Slf4j
 4public class OrderTask {
 5    @Autowired
 6    private OrderMapper orderMapper;
 7    @Scheduled(cron="0 * * * * ? ")
 8    public void processTimeoutOrder(){
 9        log.info("定时处理超时订单:{}", LocalDateTime.now());
10        LocalDateTime dueTime=LocalDateTime.now().minusMinutes(15);
11        List<Orders> timeOutOrders= orderMapper.getByStatusAndOrderTimeLT(Orders.PENDING_PAYMENT,dueTime);
12        if(timeOutOrders!=null && !timeOutOrders.isEmpty()){
13            for(Orders order :timeOutOrders){
14                order.setStatus(Orders.CANCELLED);
15                order.setCancelReason("订单超时，自动取消");
16                order.setCancelTime(LocalDateTime.now());
17                orderMapper.update(order);
18            }
19        }
20    }
21    @Scheduled(cron = "0 0 1 * * ? ")
22    public void processDeliveryOrder(){
23        log.info("自动查询派送中的订单:{}", LocalDateTime.now());
24        LocalDateTime time = LocalDateTime.now().minusMinutes(60);
25
26        List<Orders> timeOutOrders= orderMapper.getByStatusAndOrderTimeLT(Orders.PENDING_PAYMENT,time);
27        if(timeOutOrders!=null && !timeOutOrders.isEmpty()){
28            for(Orders order :timeOutOrders){
29                order.setStatus(Orders.COMPLETED);
30                order.setCancelReason("长时间处于派送中，自动完成");
31                order.setCancelTime(LocalDateTime.now());
32                orderMapper.update(order);
33            }
34        }
35    }
36}
